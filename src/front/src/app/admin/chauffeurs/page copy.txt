"use client";

import { Button } from "@/components/atoms/button";
import {
  Table,
  TableBody,
  TableCell,
  TableHead,
  TableHeader,
  TableRow,
} from "@/components/molecules/table";
import { useRouter } from "next/navigation";

import {
  DropdownMenu,
  DropdownMenuContent,
  DropdownMenuItem,
  DropdownMenuTrigger,
} from "@/components/atoms/dropdown-menu";

import { Avatar, AvatarImage, AvatarFallback } from "@/components/atoms/avatar";
import { Input } from "@/components/atoms/input";
import { ModalNames, useModalStore } from "@/store/modal";
import useGetTenants from "@/hooks/useTenants";
import { useTenantsStore } from "@/store/tenants";
import { tenantProvider } from "@/api/tenants";
import { useMemo, useState, useEffect } from "react";
import { useCheckSubscription } from "@/hooks/useAuth";
import { contractProvider } from "@/api/contracts";
import { toast } from "react-toastify";
import { conversationProvider } from "@/api/conversation";
import { useChatStore } from "@/store/chat";
import { useAuthStore } from "@/store/auth";
import {
  withBulkDelete,
  BulkDeleteProps,
  BulkDeleteResult,
} from "@/components/hoc/with-bulk-delete";

function Page({
  BulkActionBar,
  SelectAllCheckbox,
  SelectableCheckbox,
  selectedIds,
}: BulkDeleteProps) {
  const { toggle, open } = useModalStore();
  const { tenants, removeTenant } = useTenantsStore();
  const [search, setSearch] = useState("");
  const { subscribed } = useCheckSubscription(); // Get subscription status

  const router = useRouter(); // For navigation
  const [currentPage, setCurrentPage] = useState(1);
  const elementsPerPage = 10; // Number of tenants per page

  const [contracts, setContracts] = useState<any[]>([]);
  const [tenantPropertiesMap, setTenantPropertiesMap] = useState<
    Record<string, any[]>
  >({});

  const {
    conversations,
    selectedConversationId,
    loaded,
    addMessage,
    selectConversation,
    createConversation,
  } = useChatStore();

  const { user } = useAuthStore();

  const isSubscribed = subscribed && user?.validationStatus === "validate";

  useEffect(() => {
    setCurrentPage(1);
  }, [tenants]);

  useEffect(() => {
    setCurrentPage(1);
  }, []);

  // Fetch contracts and build tenant-properties mapping
  useEffect(() => {
    const fetchContracts = async () => {
      try {
        const { data } = await contractProvider.getContractsList();
        if (data?.contracts) {
          setContracts(data.contracts);
          // Build mapping: tenantId -> [property1, property2, ...]
          const map: Record<string, any[]> = {};
          data.contracts.forEach((contract: any) => {
            const tenantId = contract.tenant_id;
            const property = {
              id: contract.property_id,
              name: contract.property_name,
              address: contract.property_address,
            };
            if (!map[tenantId]) map[tenantId] = [];
            map[tenantId].push(property);
          });
          setTenantPropertiesMap(map);
        }
      } catch (error) {
        console.error("Failed to fetch contracts:", error);
      }
    };
    fetchContracts();
  }, []);

  const [isDeleting, setIsDeleting] = useState(false);

  const [filterDropdownOpen, setFilterDropdownOpen] = useState(false);
  // Track which row dropdown is open
  const [openRowDropdown, setOpenRowDropdown] = useState<number | null>(null);

  useGetTenants({ page: 1 });

  const [filters, setFilters] = useState({
    profession: "",
    logement: "",
  });

  const [draftFilters, setDraftFilters] = useState({
    profession: "",
    logement: "",
  });

  // Calculate total filtered count for pagination
  const filteredCount = useMemo(() => {
    return Array.isArray(tenants)
      ? tenants.filter((tenant) => {
          // Name filter
          const nameMatch =
            !search || tenant.name.toLowerCase().includes(search.toLowerCase());

          // Profession filter with null check
          const professionMatch =
            !filters.profession ||
            (tenant.professions &&
              tenant.professions
                .toLowerCase()
                .includes(filters.profession.toLowerCase()));

          return nameMatch && professionMatch;
        }).length
      : 0;
  }, [tenants, search, filters]);

  // Close all dropdowns when modal opens
  useEffect(() => {
    if (open) {
      setFilterDropdownOpen(false);
      setOpenRowDropdown(null);
    }
  }, [open]);

  // Initialize draft filters from actual filters when dropdown opens
  const handleDropdownOpen = () => {
    setDraftFilters({ ...filters });
  };

  // Apply filters from draft when Apply is clicked
  const applyFilters = () => {
    setFilters({ ...draftFilters });
    setCurrentPage(1);
    // Close the dropdown using the controlled state
    setFilterDropdownOpen(false);
  };

  // Clear draft filters
  const clearDraftFilters = () => {
    setDraftFilters({
      profession: "",
      logement: "",
    });
  };

  // Clear all active filters
  const clearAllFilters = () => {
    setFilters({
      profession: "",
      logement: "",
    });
    setCurrentPage(1);
  };

  const handleDeleteTenant = (tenant: any) => {
    toggle({
      name: ModalNames.CONFIRM_DELETE_TENANT,
      data: {
        title: "Supprimer le locataire",
        description: `Êtes-vous sûr de vouloir supprimer le locataire "${tenant.name}" ? Cette action est irréversible.`,
        itemName: tenant.name,
        onConfirm: async () => {
          setIsDeleting(true);
          try {
            const result = await tenantProvider.delete(tenant.id);
            if (result.data) {
              // Remove from store
              removeTenant(tenant.id);
              // Optional: Show success message
              console.log("");
            }
          } catch (error) {
            console.error("Failed to delete tenant:", error);
            // Optional: Show error message
          } finally {
            setIsDeleting(false);
          }
        },
        isLoading: isDeleting,
      },
    });
  };

  // Enhanced filtering logic
  const filteredTenants = useMemo(() => {
    const filtered = Array.isArray(tenants)
      ? tenants.filter((tenant) => {
          // Name filter
          const nameMatch =
            !search || tenant.name.toLowerCase().includes(search.toLowerCase());

          // Profession filter with null check
          const professionMatch =
            !filters.profession ||
            (tenant.professions &&
              tenant.professions
                .toLowerCase()
                .includes(filters.profession.toLowerCase()));

          // Return true only if all active filters match
          return nameMatch && professionMatch;
        })
      : [];

    // Apply pagination slicing
    return filtered.slice(
      (currentPage - 1) * elementsPerPage,
      currentPage * elementsPerPage
    );
  }, [tenants, search, filters, currentPage]);

  console.log("Here are the filtered tenants", filteredTenants);

  const handleRowDropdownChange = (index: number, isOpen: boolean) => {
    setOpenRowDropdown(isOpen ? index : null);
  };

  const handleRowActionClick = (action: () => void, index: number) => {
    // Close the dropdown first
    setOpenRowDropdown(null);
    // Then execute the action
    action();
  };

  // Pagination display function
  const displayPagination = () => {
    const pageNumber = Math.ceil(filteredCount / elementsPerPage);

    const pages = [];

    for (let i = 1; i <= pageNumber; i++) {
      pages.push(
        <Button
          key={i}
          className={`w-10 h-10 mb-[4.5rem] rounded-lg bg-transparent shadow-none ${
            currentPage === i
              ? "bg-black hover:bg-black text-white"
              : "text-black hover:bg-black/10"
          }`}
          onClick={() => setCurrentPage(i)}
        >
          {i}
        </Button>
      );
    }

    return pages;
  };

  return (
    <section>
      {!isSubscribed && (
        <div
          role="alert"
          className="w-full bg-yellow-100 border-b border-yellow-300 rounded-[12px] px-4 py-3 flex flex-col sm:flex-row items-start sm:items-center justify-between text-yellow-800 gap-2 mb-4"
        >
          <div className="flex items-center gap-2 flex-1">
            <svg
              width="20"
              height="20"
              viewBox="0 0 24 24"
              fill="none"
              xmlns="http://www.w3.org/2000/svg"
              className="stroke-yellow-600 flex-shrink-0"
            >
              <path
                d="M12 9V11M12 15H12.01M21 12C21 16.9706 16.9706 21 12 21C7.02944 21 3 16.9706 3 12C3 7.02944 7.02944 3 12 3C16.9706 3 21 7.02944 21 12Z"
                strokeWidth="1.5"
                strokeLinecap="round"
                strokeLinejoin="round"
              />
            </svg>
            <span className="text-sm font-medium leading-tight">
              Vous ne pouvez pas ajouter de locataires sans activer un
              abonnement. Activez un plan pour accéder à cette fonctionnalité.
            </span>
          </div>
          <button
            onClick={() => router.push("/admin/abonnements")}
            className="bg-yellow-600 hover:bg-yellow-700 text-white px-3 py-2 sm:px-4 rounded-[12px] text-sm font-medium transition-colors w-full sm:w-auto"
          >
            Activer un plan
          </button>
        </div>
      )}
      <div>
        <div className="flex items-center justify-between">
          <h2 className="py-4 font-MontserratBold font-lg">Liste des locataires</h2>

          <div className="relative flex items-center gap-4">
            {/* Search input */}

              <Input
                className="h-10 bg-background pl-10"
                placeholder="Search"
                value={search}
                onChange={(e) => setSearch(e.target.value)}
              />

                  height="16"
                  viewBox="0 0 24 24"
                  fill="none"
                  className="stroke-gray-500"

                    strokeLinecap="round"
                    strokeLinejoin="round"

              onOpenChange={(open) => {
                setFilterDropdownOpen(open);
                if (open) handleDropdownOpen();

            >
              <DropdownMenuTrigger asChild>

                  variant="outline"

                  <svg
                    width="16"
                    height="16"
                    viewBox="0 0 24 24"
                    fill="none"

                      stroke="currentColor"
                      strokeWidth="1.5"
                      strokeLinecap="round"
                      strokeLinejoin="round"
                    />
                  </svg>
                  Filtres

                    <span className="w-2 h-2 bg-primary rounded-full ml-1"></span>

                </Button>
              </DropdownMenuTrigger>
              <DropdownMenuContent className="w-80 p-4" align="end">
                <div className="space-y-4">
                  <div>
                    <h4 className="mb-2 text-sm font-medium">Profession</h4>
                    <Input
                      placeholder="Filtrer par profession"
                      value={draftFilters.profession}
                      onChange={(e) =>
                        setDraftFilters({
                          ...draftFilters,
                          profession: e.target.value,
                        })
                      }
                    />
                  </div>

                  <div>
                    <h4 className="mb-2 text-sm font-medium">Logement</h4>
                    <Input
                      placeholder="Filtrer par logement"
                      value={draftFilters.logement}
                      onChange={(e) =>
                        setDraftFilters({
                          ...draftFilters,
                          logement: e.target.value,
                        })
                      }
                    />
                  </div>

                  <div className="flex justify-between pt-2">
                    <Button
                      variant="outline"
                      size="sm"
                      onClick={clearDraftFilters}
                    >
                      Réinitialiser
                    </Button>
                    <Button size="sm" onClick={applyFilters}>
                      Appliquer
                    </Button>
                  </div>
                </div>
              </DropdownMenuContent>
            </DropdownMenu>

            {/* Add tenant button - Disabled if not subscribed */}
            <Button
              className={`w-auto h-10 rounded-lg ${
                !isSubscribed ? "opacity-50 cursor-not-allowed" : ""
              }`}
              disabled={!isSubscribed}
              onClick={() =>
                isSubscribed &&
                toggle({
                  name: ModalNames.ADD_LOCATAIRE,
                  data: {
                    type: "create",
                  },
                })
              }
            >
              <svg
                width="24"
                height="24"
                viewBox="0 0 24 24"
                fill="none"
                xmlns="http://www.w3.org/2000/svg"
                className="fill-primary-foreground"
              >
                <path d="M9.5999 10.8001C11.5881 10.8001 13.1999 9.18832 13.1999 7.2001C13.1999 5.21187 11.5881 3.6001 9.5999 3.6001C7.61168 3.6001 5.9999 5.21187 5.9999 7.2001C5.9999 9.18832 7.61168 10.8001 9.5999 10.8001Z" />
                <path d="M9.5999 13.2001C13.5764 13.2001 16.7999 16.4236 16.7999 20.4001H2.3999C2.3999 16.4236 5.62345 13.2001 9.5999 13.2001Z" />
                <path d="M19.1999 8.4001C19.1999 7.73736 18.6626 7.2001 17.9999 7.2001C17.3372 7.2001 16.7999 7.73736 16.7999 8.4001V9.6001H15.5999C14.9372 9.6001 14.3999 10.1374 14.3999 10.8001C14.3999 11.4628 14.9372 12.0001 15.5999 12.0001H16.7999V13.2001C16.7999 13.8628 17.3372 14.4001 17.9999 14.4001C18.6626 14.4001 19.1999 13.8628 19.1999 13.2001V12.0001H20.3999C21.0626 12.0001 21.5999 11.4628 21.5999 10.8001C21.5999 10.1374 21.0626 9.6001 20.3999 9.6001H19.1999V8.4001Z" />
              </svg>

              <span>Ajouter un locataire</span>
            </Button>
          </div>
        </div>

        <div className="w-full bg-background rounded-2xl p-4">
          {/* Filter indicators above the table */}
          {Object.values(filters).some((v) => v) && (
            <div className="mb-4 flex flex-wrap gap-2 items-center">
              <span className="text-sm font-medium mr-2">Filtres actifs:</span>
              {filters.profession && (
                <div className="bg-primary/10 text-sm px-2 py-1 rounded-md flex items-center gap-1">
                  Profession: {filters.profession}
                  <button
                    onClick={() => setFilters({ ...filters, profession: "" })}
                  >
                    <svg
                      width="14"
                      height="14"
                      viewBox="0 0 24 24"
                      fill="none"
                      xmlns="http://www.w3.org/2000/svg"
                    >
                      <path
                        d="M18 6L6 18M6 6l12 12"
                        stroke="currentColor"
                        strokeWidth="2"
                        strokeLinecap="round"
                        strokeLinejoin="round"
                      />
                    </svg>
                  </button>
                </div>
              )}
              {filters?.logement && (
                <div className="bg-primary/10 text-sm px-2 py-1 rounded-md flex items-center gap-1">
                  Logement: {filters.logement}
                  <button
                    onClick={() => setFilters({ ...filters, logement: "" })}
                  >
                    <svg
                      width="14"
                      height="14"
                      viewBox="0 0 24 24"
                      fill="none"
                      xmlns="http://www.w3.org/2000/svg"
                    >
                      <path
                        d="M18 6L6 18M6 6l12 12"
                        stroke="currentColor"
                        strokeWidth="2"
                        strokeLinecap="round"
                        strokeLinejoin="round"
                      />
                    </svg>
                  </button>
                </div>
              )}
              {Object.values(filters).some((v) => v) && (
                <button
                  className="text-sm text-primary hover:underline"
                  onClick={clearAllFilters}
                >
                  Effacer tout
                </button>
              )}
            </div>
          )}

          {/* Tenants table */}
          <Table className="min-w-[1000px] overflow-auto">
            <TableHeader>

              <TableRow className="hover:bg-transparent">
                <TableHead className="w-[50px]">
                                    <SelectAllCheckbox
                                      allIds={filteredTenants?.map((t) => t.id) ?? []}
                                    />
                                  </TableHead>
                <TableHead className="w-[250px]">Nom du locataire</TableHead>
                <TableHead>Logement</TableHead>
                <TableHead className="text-left">Numéro de téléphone</TableHead>
                <TableHead className="text-left">Profession</TableHead>
                <TableHead className="text-right">Action</TableHead>
              </TableRow>
            </TableHeader>

            <TableBody>
              {filteredTenants?.map((tenant, index) => {
                const tenantProperties = tenantPropertiesMap[tenant.id] || [];
                const firstProperty = tenantProperties[0];

                return (
                  <TableRow key={index}>
                     {/* Add Selectable Checkbox */}
                      <TableCell onClick={(e) => e.stopPropagation()}>
                        <SelectableCheckbox
                          id={tenant.id}
                          isSelected={selectedIds.has(tenant.id)}
                        />
                      </TableCell>
                    <TableCell className="py-4">
                      <div className="font-medium flex items-center gap-2">
                        <Avatar className="w-[30px] h-[30px]">
                          <AvatarImage
                            src="/static/avatar-default.webp"
                            alt={tenant.name}
                          />
                          <AvatarFallback>
                            {tenant.name.substring(0, 2).toUpperCase()}
                          </AvatarFallback>
                        </Avatar>

                        <span>{tenant.name}</span>
                      </div>
                    </TableCell>
                    <TableCell className="text-left">
                      {firstProperty
                        ? `${firstProperty?.name} - ${firstProperty?.address}`
                        : "-"}
                    </TableCell>

                    <TableCell className="text-left">{tenant.phone}</TableCell>
                    <TableCell>{tenant.professions || "-"}</TableCell>
                    <TableCell className="text-right">
                      <DropdownMenu
                        open={openRowDropdown === index}
                        onOpenChange={(isOpen) =>
                          handleRowDropdownChange(index, isOpen)
                        }
                      >
                        <DropdownMenuTrigger asChild>
                          <div className="inline-flex relative w-[44px] h-[44px] rounded-full bg-primary/10 items-center justify-center cursor-pointer hover:bg-primary/20">
                            <svg
                              width="24"
                              height="24"
                              viewBox="0 0 24 24"
                              fill="none"
                              xmlns="http://www.w3.org/2000/svg"
                            >
                              <path
                                d="M12 13C12.5523 13 13 12.5523 13 12C13 11.4477 12.5523 11 12 11C11.4477 11 11 11.4477 11 12C11 12.5523 11.4477 13 12 13Z"
                                stroke="currentColor"
                                strokeWidth="2"
                                strokeLinecap="round"
                                strokeLinejoin="round"
                              />
                              <path
                                d="M19 13C19.5523 13 20 12.5523 20 12C20 11.4477 19.5523 11 19 11C18.4477 11 18 11.4477 18 12C18 12.5523 18.4477 13 19 13Z"
                                stroke="currentColor"
                                strokeWidth="2"
                                strokeLinecap="round"
                                strokeLinejoin="round"
                              />
                              <path
                                d="M5 13C5.55228 13 6 12.5523 6 12C6 11.4477 5.55228 11 5 11C4.44772 11 4 11.4477 4 12C4 12.5523 4.44772 13 5 13Z"
                                stroke="currentColor"
                                strokeWidth="2"
                                strokeLinecap="round"
                                strokeLinejoin="round"
                              />
                            </svg>
                          </div>
                        </DropdownMenuTrigger>
                        <DropdownMenuContent align="end" className="w-48">
                          <div className="grid grid-cols-1 gap-1 p-1">
                            <DropdownMenuItem
                              onClick={() =>
                                handleRowActionClick(() => {
                                  const {
                                    id,
                                    name,
                                    phone,
                                    email,
                                    professions,
                                    profilePicture,
                                    identityDocument,
                                    ...rest
                                  } = tenant;
                                  const mappedTenant = {
                                    id,
                                    name,
                                    phone,
                                    email,
                                    professions,
                                    address: firstProperty?.address || "",
                                    profilePicture:"https://backend.mymidaas.com"+ profilePicture, // rename _ to camelCase
                                    identityDocument:"https://backend.mymidaas.com"+ identityDocument, 
                                    ...rest,
                                    properties: tenantProperties, // Pass full list of properties
                                  };

                                  toggle({
                                    name: ModalNames.VIEW_EDIT_LOCATAIRE,
                                    data: {
                                      tenant: mappedTenant,
                                      type: "view",
                                    },
                                  });
                                }, index)
                              }
                              className="cursor-pointer flex items-center gap-3 px-3 py-2.5 rounded-md hover:bg-accent hover:border-none"
                            >
                              <div className="flex items-center justify-center w-8 h-8 rounded-full bg-blue-50">
                                <svg
                                  width="16"
                                  height="16"
                                  viewBox="0 0 24 24"
                                  fill="none"
                                  xmlns="http://www.w3.org/2000/svg"

                                    d="M2 12C2 12 5 5 12 5C19 5 22 12 22 12C22 12 19 19 12 19C5 19 2 12 2 12Z"
                                    stroke="currentColor"

                                  />
                                  <path
                                    d="M12 15C13.6569 15 15 13.6569 15 12C15 10.3431 13.6569 9 12 9C10.3431 9 9 10.3431 9 12C9 13.6569 10.3431 15 12 15Z"
                                    stroke="currentColor"
                                    strokeWidth="2"
                                    strokeLinecap="round"
                                    strokeLinejoin="round"
                                  />
                                </svg>
                              </div>
                              <span className="text-sm font-medium">
                                Voir les détails

                            </DropdownMenuItem>

                            <DropdownMenuItem
                              onClick={() =>
                                handleRowActionClick(() => {
                                  const {
                                    id,
                                    name,
                                    phone,
                                    email,
                                    professions,
                                    profilePicture,

                                    ...rest
                                  } = tenant;
                                  const mappedTenant = {
                                    id,
                                    name,
                                    phone,
                                    email,
                                    professions,
                                    address: firstProperty?.address || "",
                                    profilePicture:"https://backend.mymidaas.com"+ profilePicture, // rename _ to camelCase

                                    ...rest,
                                    properties: tenantProperties, // Pass full list of properties
                                  };

                                  toggle({
                                    name: ModalNames.VIEW_EDIT_LOCATAIRE,
                                    data: {
                                      tenant: mappedTenant,
                                      type: "edit",
                                    },
                                  });
                                }, index)
                              }
                              className="cursor-pointer flex items-center gap-3 px-3 py-2.5 rounded-md hover:bg-accent hover:border-none"
                            >
                              <div className="flex items-center justify-center w-8 h-8 rounded-full bg-green-50">
                                <svg
                                  width="16"
                                  height="16"
                                  viewBox="0 0 24 24"
                                  fill="none"
                                  xmlns="http://www.w3.org/2000/svg"
                                  className="text-green-600"
                                >
                                  <path
                                    d="M11 4H4C3.46957 4 2.96086 4.21071 2.58579 4.58579C2.21071 4.96086 2 5.46957 2 6V20C2 20.5304 2.21071 21.0391 2.58579 21.4142C2.96086 21.7893 3.46957 22 4 22H18C18.5304 22 19.0391 21.7893 19.4142 21.4142C19.7893 21.0391 20 20.5304 20 20V13"

                                    d="M18.5 2.50001C18.8978 2.10219 19.4374 1.87869 20 1.87869C20.5626 1.87869 21.1022 2.10219 21.5 2.50001C21.8978 2.89784 22.1213 3.4374 22.1213 4.00001C22.1213 4.56262 21.8978 5.10219 21.5 5.50001L12 15L8 16L9 12L18.5 2.50001Z"

export default withBulkDelete({
  onDeleteItems: async (ids) => {
    const results = await Promise.allSettled(
      ids.map((id) => tenantProvider.delete(id.toString()))
    );

    const successfulIds: (string | number)[] = [];
    const failedIds: (string | number)[] = [];
    const errors: { id: string | number; error: any }[] = [];

    results.forEach((result, index) => {
      const id = ids[index];
      if (result.status === "fulfilled") {
        successfulIds.push(id);
      } else {
        failedIds.push(id);
        errors.push({ id, error: result.reason });
      }
    });

    return { successfulIds, failedIds, errors };
  },
  storeMethods: {
    removeTenant: useTenantsStore.getState().removeTenant,
    // refreshTenants: useTenantsStore.getState().refreshTenants
  },
  onSuccess: async (result, storeMethods) => {
    if (result.successfulIds.length > 0) {
      // Remove each tenant from store
      result.successfulIds.forEach(id => {
        storeMethods.removeTenant?.(id.toString());
      });
      // Refresh the list
      await storeMethods.refreshTenants?.();
      toast.success(
        `${result.successfulIds.length} locataire(s) supprimé(s) avec succès`
      );
    }
    if (result.failedIds.length > 0) {
      toast.error(
        `Échec de suppression pour ${result.failedIds.length} locataire(s)`
      );
    }
  },
  confirmMessage: "Êtes-vous sûr de vouloir supprimer les locataires sélectionnés?",
  allowPartialSuccess: true,
})(Page);
